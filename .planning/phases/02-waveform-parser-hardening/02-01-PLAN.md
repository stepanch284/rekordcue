```python
def get_pssi_sections(db, content) -> list | None:
    """Read PSSI (Pioneer Segmentation Slice Info) phrase tags from .EXT file.

    Returns list of (start_beat, end_beat, phrase_type) tuples if available, else None.
    This is Rekordbox's own phrase analysis -- more reliable than custom energy detection.
    """
    ext_path = db.get_anlz_path(content, 'EXT')
    if not ext_path or not Path(ext_path).exists():
        return None  # .EXT file doesn't exist

    try:
        anlz_ext = AnlzFile.parse_file(ext_path)
        if 'PHRS' in anlz_ext.tag_types:
            phrs_tag = anlz_ext.get_tag('PHRS')
            phrs_data = phrs_tag.get()
            if phrs_data and len(phrs_data) > 0:
                return phrs_data
    except Exception as e:
        # Log but don't crash -- PSSI is optional, PWAV fallback available
        pass

    return None


def get_pwav_amplitudes_safe(db, content) -> np.ndarray | None:
    """Read PWAV amplitude data with comprehensive error handling.

    Returns amplitude array or None if any error occurs (doesn't crash).
    Prints user-visible warning for each failure scenario.
    """
    dat_path = db.get_anlz_path(content, 'DAT')

    if not dat_path:
        print(f"[WARN] Track {content.ID}: No .DAT file path. Skipping.")
        return None

    if not Path(dat_path).exists():
        print(f"[WARN] Track {content.ID}: .DAT file not found. Skipping.")
        return None

    try:
        anlz = AnlzFile.parse_file(dat_path)
        if 'PWAV' not in anlz.tag_types:
            print(f"[WARN] Track {content.ID}: No PWAV tag in .DAT. Skipping.")
            return None
        pwav_data = anlz.get_tag('PWAV').get()[0]
        return pwav_data.astype(float)

    except Exception as e:
        print(f"[WARN] Track {content.ID}: ANLZ parse error ({type(e).__name__}). Skipping.")
        return None
```
