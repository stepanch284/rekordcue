```python
def compute_rms_per_bar(bar_energies: np.ndarray) -> np.ndarray:
    """Compute RMS (Root Mean Square) of energy derivatives per bar.

    High RMS = sharp transitions (drops, breakdowns).
    Low RMS = smooth, steady regions (intro, sustained high energy).
    """
    diffs = np.abs(np.diff(bar_energies))
    # Window RMS: compute RMS over ~4-bar windows
    rms_arr = np.array([np.sqrt(np.mean(diffs[max(0, i-2):i+2]**2))
                         for i in range(len(bar_energies))])
    return rms_arr

def detect_sections_rms_hybrid(bar_energies: np.ndarray,
                               bar_times_ms: np.ndarray,
                               intro_bars: int = 16) -> dict:
    """Hybrid detection using mean energy + RMS variance.

    Returns: same dict format as detect_sections_from_energy()
    """
    intro_baseline = np.mean(bar_energies[0:intro_bars])
    high_threshold = intro_baseline * 1.8
    low_threshold = intro_baseline * 0.6

    rms_arr = compute_rms_per_bar(bar_energies)
    rms_baseline = np.mean(rms_arr[0:intro_bars])

    sections = {}

    # Simple scanning (same as Simple Threshold, but with RMS validation)
    drop1_found = False
    breakdown_found = False

    for bar_idx in range(intro_bars, len(bar_energies)):
        energy = bar_energies[bar_idx]
        rms = rms_arr[bar_idx] if bar_idx < len(rms_arr) else 0

        # Drop 1: high energy AND sharp transition (RMS spike)
        if not drop1_found and energy > high_threshold and rms > rms_baseline * 1.3:
            sections['drop1_bar'] = bar_idx
            drop1_found = True

        # Breakdown: low energy but also low RMS (smooth valley)
        elif drop1_found and not breakdown_found and energy < low_threshold and rms < rms_baseline * 0.8:
            valley_duration = 0
            for j in range(bar_idx, min(bar_idx + 8, len(bar_energies))):
                if bar_energies[j] < low_threshold:
                    valley_duration += 1
            if valley_duration >= 2:
                sections['breakdown_bar'] = bar_idx
                breakdown_found = True

        # Drop 2: high energy again with RMS spike
        elif breakdown_found and energy > high_threshold and rms > rms_baseline * 1.3:
            sections['drop2_bar'] = bar_idx
            break

    # Outro
    outro_energy = np.mean(bar_energies[-16:])
    if outro_energy < intro_baseline * 1.2:
        sections['outro_bar'] = max(len(bar_energies) - 16, 0)

    # Confidence scoring (same as Simple Threshold version)
    for section_type in ['drop1', 'breakdown', 'drop2', 'outro']:
        bar_key = f'{section_type}_bar'
        conf_key = f'{section_type}_confidence'
        if sections.get(bar_key) is not None:
            sections[conf_key] = compute_section_confidence(
                bar_energies, sections[bar_key], section_type
            )

    return sections
```
